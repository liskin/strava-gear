Common invocation flags:

    $ function strava-gear {
    >   command strava-gear --rules rules.yaml --csv - --tablefmt csv "$@"
    > }

Rules:

    $ cat >rules.yaml <<END
    > rules:
    >   - road:
    >       frame: fr
    >       chain: c1
    >   - since: 2023-01-01
    >     road:
    >       chain: c2
    >   - since: 2024-01-01
    >     road:
    >       chain: c3
    >   - '#name=Ride 3':
    >       chain: c4
    >   - '#type=VirtualRide':
    >       chain: c5
    > END
